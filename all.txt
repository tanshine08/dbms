7. INNER JOIN
Question

Retrieve customer names with order details using INNER JOIN.

Complete Answer
CREATE TABLE Customer(
    CustomerID INT PRIMARY KEY,
    Name VARCHAR(50)
);

CREATE TABLE Orders(
    OrderID INT PRIMARY KEY,
    CustomerID INT,
    Amount INT,
    FOREIGN KEY(CustomerID) REFERENCES Customer(CustomerID)
);

INSERT INTO Customer VALUES
(1,'Tanmay'),
(2,'Rahul');

INSERT INTO Orders VALUES
(101,1,5000),
(102,2,7000);

SELECT Customer.Name, Orders.OrderID, Orders.Amount
FROM Customer
INNER JOIN Orders
ON Customer.CustomerID = Orders.CustomerID;
8. LEFT JOIN
Question

Complete Answer

FROM Customer

ON Customer.CustomerID = Orders.CustomerID;

10. Trigger
Question

Create a trigger that records salary updates in Employee_Audit table.

Complete Answer
CREATE TABLE Employee(
    EmpID INT PRIMARY KEY,
    Name VARCHAR(50),
    Salary INT
);

CREATE TABLE Employee_Audit(
    EmpID INT,
    OldSalary INT,
    NewSalary INT
);

DELIMITER //

CREATE TRIGGER Salary_Update
BEFORE UPDATE ON Employee
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Audit
    VALUES(OLD.EmpID, OLD.Salary, NEW.Salary);
END//

DELIMITER ;

INSERT INTO Employee VALUES
(1,'Tanmay',30000);

UPDATE Employee
SET Salary = 40000
WHERE EmpID = 1;

SELECT * FROM Employee_Audit;

16. Procedure Even or Odd
Question

Write procedure to check even or odd number.

Complete Answer
DELIMITER //

CREATE PROCEDURE EvenOdd(IN num INT)
BEGIN

    IF num % 2 = 0 THEN
        SELECT 'Even Number';
    ELSE
        SELECT 'Odd Number';
    END IF;

END//

DELIMITER ;

CALL EvenOdd(10);
17. Function Greater Number
Question

Write function to find greater of two numbers.

Complete Answer
DELIMITER //

CREATE FUNCTION GreaterNum(a INT,b INT)
RETURNS INT
DETERMINISTIC
BEGIN

    IF a>b THEN
        RETURN a;
    ELSE
        RETURN b;
    END IF;

END//

DELIMITER ;

SELECT GreaterNum(10,20);

20. Display Even Numbers 1 to 20
Question

Write procedure to display even numbers between 1 to 20.

Complete Answer
DELIMITER //

CREATE PROCEDURE EvenNumbers()
BEGIN

    DECLARE i INT DEFAULT 1;

    WHILE i <= 20 DO

        IF i % 2 = 0 THEN
            SELECT i;
        END IF;

        SET i = i + 1;

    END WHILE;

END//

DELIMITER ;

CALL EvenNumbers();
21. Display Odd Numbers Using Continue
Question

Write procedure to display odd numbers between 1 to 20 using continue.

Complete Answer
DELIMITER //

CREATE PROCEDURE OddNumbers()
BEGIN

    DECLARE i INT DEFAULT 0;

    myloop: LOOP

        SET i = i + 1;

        IF i > 20 THEN
            LEAVE myloop;
        END IF;

        IF i % 2 = 0 THEN
            ITERATE myloop;
        END IF;

        SELECT i;

    END LOOP;

END//

DELIMITER ;

CALL OddNumbers();
